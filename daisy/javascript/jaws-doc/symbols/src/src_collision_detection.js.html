 15 /**
 16  * collides 2 single objects by reading x, y and either method rect() or property radius.
 17  * returns true if the two objects are colliding.
 18  */
 19 jaws.collideOneWithOne = function(object1, object2) {
 20   if(object1.radius && object2.radius && object1 !== object2 && jaws.collideCircles(object1, object2))          return true;
 21   if(object1.rect && object2.rect && object1 !== object2 && jaws.collideRects( object1.rect(), object2.rect())) return true;
 22   return false;
 23 }

 25 /**
 26  * collide one single object 'object' with a list of objects 'list'.
 27  * returns an array of items from 'list' that collided with 'object'.
 28  * returns empty array of no collisions are found.
 29  * will never collide objects with themselves.
 30  */
 31 jaws.collideOneWithMany = function(object, list) {
 32   return list.filter( function(item) { return jaws.collideOneWithOne(object, item) } ) 
 33 }
 34 
 35 /**
 36  * Collides two list/arrays of objects -- 'list1' and 'list2'.
 37  * Returns an array of arrays with colliding pairs from 'list1' and 'list2'.
 38  * Will never collide objects with themselves, even if you collide the same list with itself.
 39  *
 40  * @example
 41  *
 42  *   jaws.collideManyWithMany(bullets, enemies) // --> [[bullet, enemy], [bullet, enemy]]
 43  *
 44  */
 45 jaws.collideManyWithMany = function(list1, list2) {
 46   var a = []
 47 
 48   if(list1 === list2) {
 49     combinations(list1, 2).forEach( function(pair) {
 50       if( jaws.collideOneWithOne(pair[0], pair[1]) ) a.push([pair[0], pair[1]]);
 51     });
 52   }
 53   else {
 54     list1.forEach( function(item1) { 
 55       list2.forEach( function(item2) { 
 56         if(jaws.collideOneWithOne(item1, item2)) a.push([item1, item2])
 57       });
 58     });
 59   }
 60 
 61   return a;
 62 }

 65 /** 
 66  * Returns true if circles collide.
 67  * Takes two objects with properties "radius" as argument.
 68  */
 69 jaws.collideCircles = function(object1, object2) {
 70   return ( jaws.distanceBetween(object1, object2) < object1.radius + object2.radius )
 71 }

 73 /** 
 74  * Returns true if 'rect1' collides with 'rect2'
 75  */
 76 jaws.collideRects = function(rect1, rect2) {
 77   return ((rect1.x >= rect2.x && rect1.x <= rect2.right) || (rect2.x >= rect1.x && rect2.x <= rect1.right)) &&
 78          ((rect1.y >= rect2.y && rect1.y <= rect2.bottom) || (rect2.y >= rect1.y && rect2.y <= rect1.bottom))
 79 }
 80 
 81 /** 
 82  * returns the distance between 2 objects
 83  */
 84 jaws.distanceBetween = function(object1, object2) {
 85   return Math.sqrt( Math.pow(object1.x-object2.x, 2) +  Math.pow(object1.y-object2.y, 2) )
 86 }

 88 /** private */
 89 function combinations(list, n) {
 90   var f = function(i) {
 91     if(list.isSpriteList !== undefined) {
 92       return list.at(i)
 93     } else {  // s is an Array
 94       return list[i];
 95     }
 96   };
 97   var r = [];
 98   var m = new Array(n);
 99   for (var i = 0; i < n; i++) m[i] = i; 
100   for (var i = n - 1, sn = list.length; 0 <= i; sn = list.length) {
101     r.push( m.map(f) );
102     while (0 <= i && m[i] == sn - 1) { i--; sn--; }
103     if (0 <= i) { 
104       m[i] += 1;
105       for (var j = i + 1; j < n; j++) m[j] = m[j-1] + 1;
106       i = n - 1;
107     }
108   }
109   return r;
110 }
