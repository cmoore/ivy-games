 45 /**
 46  * Unpacks Jaws core-constructors into the global namespace. If a global property is allready taken, a warning will be written to jaws log.
 47  * After calling jaws.unpack() you can use <b>Sprite()</b> instead of <b>jaws.Sprite()</b>, <b>Animation()</b> instead of <b>jaws.Animation()</b> and so on.
 48  *
 49  */
 50 jaws.unpack = function() {
 51   var make_global = ["Sprite", "SpriteList", "Animation", "Viewport", "SpriteSheet", "Parallax", "TileMap", "Rect", "pressed"]
 52 
 53   make_global.forEach( function(item, array, total) {
 54     if(window[item])  { jaws.log(item + "already exists in global namespace") }
 55     else              { window[item] = jaws[item] }
 56   });

 60 /**
 61  * Logs <b>msg</b> to previously found or created <div id="jaws-log">
 62  * if <b>append</b> is true, append rather than overwrite the last log-msg.
 63  */
 64 jaws.log = function(msg, append) {
 65   if(log_tag) {
 66     msg += "<br />"
 67     if(append) { log_tag.innerHTML = log_tag.innerHTML.toString() + msg } 
 68     else { log_tag.innerHTML = msg }
 69   }

 72 /**
 73  * @example
 74  * Initializes / creates:
 75  * jaws.canvas, jaws.context & jaws.dom   // our drawable gamearea
 76  * jaws.width & jaws.height               // width/height of drawable gamearea
 77  * jaws.url_parameters                    // hash of key/values of all parameters in current url
 78  * title & log_tag                        // used internally by jaws
 79  *
 80  * @private
 81  */
 82 jaws.init = function(options) {
 83   /* Find <title> tag */
 84   title = document.getElementsByTagName('title')[0]
 85   jaws.url_parameters = jaws.getUrlParameters()
 86 
 87   /*
 88    * If debug=1 parameter is present in the URL, let's either find <div id="jaws-log"> or create the tag.
 89    * jaws.log(message) will use this div for debug/info output to the gamer or developer
 90    *
 91    */
 92   log_tag = document.getElementById('jaws-log')
 93   if(jaws.url_parameters["debug"]) {
 94     if(!log_tag) {
 95       log_tag = document.createElement("div")
 96       log_tag.id = "jaws-log"
 97       log_tag.style.cssText = "overflow: auto; color: #aaaaaa; width: 300px; height: 150px; margin: 40px auto 0px auto; padding: 5px; border: #444444 1px solid; clear: both; font: 10px verdana; text-align: left;"
 98       document.body.appendChild(log_tag)
 99     }
100   }
101 
102   jaws.canvas = document.getElementsByTagName('canvas')[0]
103   if(!jaws.canvas) { jaws.dom = document.getElementById("canvas") }
104 
105   // Ordinary <canvas>, get context
106   if(jaws.canvas) { jaws.context = jaws.canvas.getContext('2d'); }
107 
108   // div-canvas / hml5 sprites, set position relative to have sprites with position = "absolute" stay within the canvas
109   else if(jaws.dom) { jaws.dom.style.position = "relative"; }  
110 
111   // Niether <canvas> or <div>, create a <canvas> with specified or default width/height
112   else {
113     jaws.canvas = document.createElement("canvas")
114     jaws.canvas.width = options.width
115     jaws.canvas.height = options.height
116     jaws.context = jaws.canvas.getContext('2d')
117     document.body.appendChild(jaws.canvas)
118   }
119 
120   
121   jaws.width = jaws.canvas ? jaws.canvas.width : jaws.dom.offsetWidth
122   jaws.height = jaws.canvas ? jaws.canvas.height  : jaws.dom.offsetHeight
123 
124   jaws.mouse_x = 0
125   jaws.mouse_y = 0
126   window.addEventListener("mousemove", saveMousePosition)
127 }
128 /**
129  * @private
130  * Keeps updated mouse coordinates in jaws.mouse_x / jaws.mouse_y
131  * This is called each time event "mousemove" triggers.
132  */
133 function saveMousePosition(e) {
134   jaws.mouse_x = (e.pageX || e.clientX)
135   jaws.mouse_y = (e.pageY || e.clientX)
136   
137   var game_area = jaws.canvas ? jaws.canvas : jaws.dom
138   jaws.mouse_x -= game_area.offsetLeft
139   jaws.mouse_y -= game_area.offsetTop
140 }

142 /** 
143  * Quick and easy startup of a jaws game loop. 
144  *
145  * @example
146  *
147  *  // jaws.start(YourGameState) It will do the following:
148  *  //
149  *  // 1) Call jaws.init() that will detect any canvas-tag (or create one for you) and set up the 2D context, then available in jaws.canvas and jaws.context.
150  *  //
151  *  // 2) Pre-load all defined assets with jaws.assets.loadAll() while showing progress, then available in jaws.assets.get("your_asset.png").
152  *  //
153  *  // 3) Create an instance of YourGameState() and call setup() on that instance. In setup() you usually create your gameobjects, sprites and so on.
154  *  // 
155  *  // 4) Loop calls to update() and draw() with given FPS (default 60) until game ends or another game state is activated.
156  *
157  *
158  *  jaws.start(MyGame)            // Start game state Game() with default options
159  *  jaws.start(MyGame, {fps: 30}) // Start game state Geme() with options, in this case jaws will run your game with 30 frames per second.
160  *  jaws.start(window)            // Use global functions setup(), update() and draw() if available. Not the recommended way but useful for testing and mini-games.
161  *
162  *  // It's recommended not giving fps-option to jaws.start since then it will default to 60 FPS and using requestAnimationFrame when possible.
163  *
164  */
165 jaws.start = function(game_state, options,game_state_setup_options) {
166   if(!options) options = {};
167   var fps = options.fps || 60
168   if (options.loading_screen === undefined)
169     options.loading_screen = true
170   
171   if(!options.width) options.width = 500; 
172   if(!options.height) options.height = 300;
173   jaws.init(options)
174 
175   displayProgress(0)
176   jaws.log("setupInput()", true)
177   jaws.setupInput()
178 
179   function displayProgress(percent_done) {
180     if(jaws.context && options.loading_screen) {
181       jaws.context.save()
182       jaws.context.fillStyle  = "black"
183       jaws.context.fillRect(0, 0, jaws.width, jaws.height);
184       jaws.context.textAlign  = "center"
185       jaws.context.fillStyle  = "white"
186       jaws.context.font       = "15px terminal";
187       jaws.context.fillText("Loading", jaws.width/2, jaws.height/2-30);
188       jaws.context.font       = "bold 30px terminal";
189       jaws.context.fillText(percent_done + "%", jaws.width/2, jaws.height/2);
190       jaws.context.restore()
191     }
192   }
193   /* Callback for when one single assets has been loaded */
194   function assetLoaded(src, percent_done) {
195     jaws.log( percent_done + "%: " + src, true)    
196     displayProgress(percent_done)
197   }
198 
199   /* Callback for when an asset can't be loaded*/
200   function assetError(src) {
201     jaws.log( "Error loading: " + src, true)
202   }
203 
204   /* Callback for when all assets are loaded */
205   function assetsLoaded() {
206     jaws.log("all assets loaded", true)
207     jaws.switchGameState(game_state||window, {fps: fps},game_state_setup_options)
208   }
209 
210   jaws.log("assets.loadAll()", true)
211   if(jaws.assets.length() > 0)  { jaws.assets.loadAll({onload:assetLoaded, onerror:assetError, onfinish:assetsLoaded}) }
212   else                          { assetsLoaded() } 
213 }

215 /**
216 * Switch to a new active game state
217 * Save previous game state in jaws.previous_game_state
218 *
219 * @example
220 * 
221 * function MenuState() {
222 *   this.setup = function() { ... }
223 *   this.draw = function() { ... }
224 *   this.update = function() {
225 *     if(pressed("enter")) jaws.switchGameState(GameState); // Start game when Enter is pressed
226 *   }
227 * }
228 *
229 * function GameState() {
230 *   this.setup = function() { ... }
231 *   this.update = function() { ... }
232 *   this.draw = function() { ... }
233 * }
234 *
235 * jaws.start(MenuState)
236 *
237 */
238 jaws.switchGameState = function(game_state, options,game_state_setup_options) {
239   var fps = (options && options.fps) || (jaws.game_loop && jaws.game_loop.fps) || 60
240   
241   jaws.game_loop && jaws.game_loop.stop()
242   jaws.clearKeyCallbacks() // clear out all keyboard callbacks
243   if(jaws.isFunction(game_state)) { game_state = new game_state }
244   
245   jaws.previous_game_state = jaws.game_state
246   jaws.game_state = game_state
247   jaws.game_loop = new jaws.GameLoop(game_state, {fps: fps},game_state_setup_options)
248   jaws.game_loop.start()
249 }

251 /** 
252  * Takes an image, returns a canvas-element containing that image.
253  * Benchmarks has proven canvas to be faster to work with then images in certain browsers.
254  * Returns: a canvas-element
255  */
256 jaws.imageToCanvas = function(image) {
257   var canvas = document.createElement("canvas")
258   canvas.src = image.src        // Make canvas look more like an image
259   canvas.width = image.width
260   canvas.height = image.height
261 
262   var context = canvas.getContext("2d")
263   context.drawImage(image, 0, 0, image.width, image.height)
264   return canvas
265 }

316 /**
317  * Returns true if <b>item</b> is outside the canvas.
318  * <b>item</b> needs to have the properties x, y, width & height

324 /**
325  * Force <b>item</b> inside canvas by setting items x/y parameters
326  * <b>item</b> needs to have the properties x, y, width & height
327  */
328 jaws.forceInsideCanvas = function(item) {
329   if(item.x < 0)              { item.x = 0  }
330   if(item.x > jaws.width)     { item.x = jaws.width }
331   if(item.y < 0)              { item.y = 0 }
332   if(item.y > jaws.height)    { item.y = jaws.height }
333 }

335 /**
336  * Return a hash of url-parameters and their values
337  *
338  * @example
339  *   // Given the current URL is <b>http://test.com/?debug=1&foo=bar</b>
340  *   jaws.getUrlParameters() // --> {debug: 1, foo: bar}
341  */
342 jaws.getUrlParameters = function() {
343   var vars = [], hash;
344   var hashes = window.location.href.slice(window.location.href.indexOf('?') + 1).split('&');
345   for(var i = 0; i < hashes.length; i++) {
346     hash = hashes[i].split('=');
347     vars.push(hash[0]);
348     vars[hash[0]] = hash[1];
349   }
350   return vars;
351 }
