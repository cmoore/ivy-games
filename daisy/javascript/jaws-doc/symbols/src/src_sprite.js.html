  3 /**
  4 * @class A basic but powerfull sprite for all your onscreen-game objects. "Field Summary" contains options for the Sprite()-constructor.
  5 * @constructor
  6 *  
  7 * @property {int} x     Horizontal position  (0 = furthest left)
  8 * @property {int} y     Vertical position    (0 = top)
  9 * @property {image} image   Image/canvas or string pointing to an asset ("player.png")
 10 * @property {int} alpha     Transparency 0=fully transparent, 1=no transperency
 11 * @property {int} angle     Angle in degrees (0-360)
 12 * @property {bool} flipped    Flip sprite horizontally, usefull for sidescrollers
 13 * @property {string} anchor   String stating how to anchor the sprite to canvas, @see Sprite#anchor ("top_left", "center" etc)
 14 * @property {int} scale_image Scale the sprite by this factor
 15 *
 16 * @example
 17 * // create new sprite at top left of the screen, will use jaws.assets.get("foo.png")
 18 * new Sprite({image: "foo.png", x: 0, y: 0}) 
 19 * 
 20 * // sets anchor to "center" on creation
 21 * new Sprite({image: "topdownspaceship.png", anchor: "center"})
 22 *
 23 */
 24 jaws.Sprite = function Sprite(options) {
 25   if( !(this instanceof arguments.callee) ) return new arguments.callee( options );
 26 
 27   this.options = options
 28   this.set(options)  
 29   
 30   if(options.context) { 
 31     this.context = options.context
 32   }
 33   else if(options.dom) {  // No canvas context? Switch to DOM-based spritemode
 34     this.dom = options.dom
 35     this.createDiv() 
 36   }
 37   if(!options.context && !options.dom) {                  // Defaults to jaws.context or jaws.dom
 38     if(jaws.context)  this.context = jaws.context;
 39     else {
 40       this.dom = jaws.dom;
 41       this.createDiv() 
 42     }
 43   }
 44 }

 46 /** 
 47  * @private
 48  * Call setters from JSON object. Used to parse options.
 49  */
 50 jaws.Sprite.prototype.set = function(options) {
 51   this.scale_x = this.scale_y = (options.scale || 1)
 52   this.x = options.x || 0
 53   this.y = options.y || 0
 54   this.alpha = (options.alpha === undefined) ? 1 : options.alpha
 55   this.angle = options.angle || 0
 56   this.flipped = options.flipped || false
 57   this.anchor(options.anchor || "top_left");
 58   if(options.anchor_x !== undefined) this.anchor_x = options.anchor_x;
 59   if(options.anchor_y !== undefined) this.anchor_y = options.anchor_y; 
 60   options.image && this.setImage(options.image);
 61   this.image_path = options.image;
 62   if(options.scale_image) this.scaleImage(options.scale_image);
 63   this.cacheOffsets()
 64 
 65   return this
 66 }

 68 /** 
 69  * @private
 70  *
 71  * Creates a new sprite from current sprites attributes()
 72  * Checks JawsJS magic property '_constructor' when deciding with which constructor to create it
 73  *
 74  */
 75 jaws.Sprite.prototype.clone = function(object) {
 76   var constructor = this._constructor ? eval(this._constructor) : this.constructor
 77   var new_sprite = new constructor( this.attributes() );
 78   new_sprite._constructor = this._constructor || this.constructor.name
 79   return new_sprite

 83 /**
 84  * Sets image from image/canvas or asset-string ("foo.png")
 85  * If asset isn't previously loaded setImage() will try to load it.
 86  */
 87 jaws.Sprite.prototype.setImage =      function(value) { 
 88   var that = this
 89 
 90   // An image, great, set this.image and return
 91   if(jaws.isDrawable(value)) {
 92     this.image = value
 93     return this.cacheOffsets() 
 94   }
 95   // Not an image, therefore an asset string, i.e. "ship.bmp"
 96   else {
 97     // Assets already loaded? Set this.image
 98     if(jaws.assets.isLoaded(value)) { this.image = jaws.assets.get(value); this.cacheOffsets(); }
 99 
100     // Not loaded? Load it with callback to set image.
101     else { jaws.assets.load(value, function() { that.image = jaws.assets.get(value); that.cacheOffsets(); }) }
102   }
103   return this
104 }

162 /**
163 * The sprites anchor could be describe as "the part of the sprite will be placed at x/y"
164 * or "when rotating, what point of the of the sprite will it rotate round"
165 *
166 * @example
167 * For example, a topdown shooter could use anchor("center") --> Place middle of the ship on x/y
168 * .. and a sidescroller would probably use anchor("center_bottom") --> Place "feet" at x/y
169 */
170 jaws.Sprite.prototype.anchor = function(value) {
171   var anchors = {
172     top_left: [0,0],
173     left_top: [0,0],
174     center_left: [0,0.5],
175     left_center: [0,0.5],
176     bottom_left: [0,1],
177     left_bottom: [0,1],
178     top_center: [0.5,0],
179     center_top: [0.5,0],
180     center_center: [0.5,0.5],
181     center: [0.5,0.5],
182     bottom_center: [0.5,1],
183     center_bottom: [0.5,1],
184     top_right: [1,0],
185     right_top: [1,0],
186     center_right: [1,0.5],
187     right_center: [1,0.5],
188     bottom_right: [1,1],
189     right_bottom: [1,1]
190   }
191 
192   if(a = anchors[value]) {
193     this.anchor_x = a[0]
194     this.anchor_y = a[1]
195     if(this.image) this.cacheOffsets();
196   }
197   return this
198 }

200 /** @private */
201 jaws.Sprite.prototype.cacheOffsets = function() {
202   if(!this.image) { return }
203   
204   this.width = this.image.width * this.scale_x
205   this.height = this.image.height * this.scale_y
206   this.left_offset   = this.width * this.anchor_x
207   this.top_offset    = this.height * this.anchor_y
208   this.right_offset  = this.width * (1.0 - this.anchor_x)
209   this.bottom_offset = this.height * (1.0 - this.anchor_y)
210 
211   if(this.cached_rect) this.cached_rect.resizeTo(this.width, this.height);
212   return this
213 }

215 /** Returns a jaws.Rect() perfectly surrouning sprite. Also cache rect in this.cached_rect. */
216 jaws.Sprite.prototype.rect = function() {
217   if(!this.cached_rect) this.cached_rect = new jaws.Rect(this.x, this.top, this.width, this.height)
218   this.cached_rect.moveTo(this.x - this.left_offset, this.y - this.top_offset)
219   return this.cached_rect
220 }

222 /**
223  * Make this sprite a DOM-based <div> sprite 
224  * @private
225  */
226 jaws.Sprite.prototype.createDiv = function() {
227   this.div = document.createElement("div")
228   this.div.style.position = "absolute"
229   if(this.image) {
230     this.div.style.width = this.image.width + "px"
231     this.div.style.height = this.image.height + "px"
232     if(this.image.toDataURL)  { this.div.style.backgroundImage = "url(" + this.image.toDataURL() + ")" }
233     else                      { this.div.style.backgroundImage = "url(" + this.image.src + ")" }
234   }
235   if(this.dom) { this.dom.appendChild(this.div) }
236   this.updateDiv()
237 }

239 /** 
240  * @private
241  * Update properties for DOM-based sprite 
242  */
243 jaws.Sprite.prototype.updateDiv = function() {
244   this.div.style.left = this.x + "px"
245   this.div.style.top = this.y + "px"
246 
247   var transform = ""
248   transform += "rotate(" + this.angle + "deg) "
249   if(this.flipped)  { transform += "scale(-" + this.scale_x + "," + this.scale_y + ")"; }
250   else              { transform += "scale(" + this.scale_x + "," + this.scale_y + ")"; }
251 
252   this.div.style.MozTransform = transform
253   this.div.style.WebkitTransform = transform
254   this.div.style.OTransform = transform
255   this.div.style.msTransform = transform
256   this.div.style.transform = transform
257 
258   return this
259 }

261 /** Draw sprite on active canvas or update it's DOM-properties */
262 jaws.Sprite.prototype.draw = function() {
263   if(!this.image) { return this }
264   if(this.dom)    { return this.updateDiv() }
265 
266   this.context.save()
267   this.context.translate(this.x, this.y)
268   if(this.angle!=0) { jaws.context.rotate(this.angle * Math.PI / 180) }
269   this.flipped && this.context.scale(-1, 1)
270   this.context.globalAlpha = this.alpha
271   this.context.translate(-this.left_offset, -this.top_offset) // Needs to be separate from above translate call cause of flipped
272   this.context.drawImage(this.image, 0, 0, this.width, this.height)
273   this.context.restore()
274   return this
275 }

320 /** returns Sprites state/properties as a pure object */
321 jaws.Sprite.prototype.attributes = function() { 
322   var object = this.options                   // Start with all creation time properties
323   object["_constructor"] = this._constructor || "jaws.Sprite"
324   object["x"] = parseFloat(this.x.toFixed(2))
325   object["y"] = parseFloat(this.y.toFixed(2))
326   object["image"] = this.image_path
327   object["alpha"] = this.alpha
328   object["flipped"] = this.flipped
329   object["angle"] = parseFloat(this.angle.toFixed(2))
330   object["scale_x"] = this.scale_x;
331   object["scale_y"] = this.scale_y;
332   object["anchor_x"] = this.anchor_x
333   object["anchor_y"] = this.anchor_y
334 
335   return object
336 }
