1 var jaws = (function(jaws) {
  2 
  3 /**
  4  *
  5  * @class A window (Rect) into a bigger canvas/image. Viewport is always contained within that given image (called the game world). "Field Summary" contains options for the Viewport()-constructor.
  6  *
  7  * @property {int} width  Width of viewport, defaults to canvas width
  8  * @property {int} height Height of viewport, defaults to canvas height
  9  * @property {int} max_x  Maximum x-position for viewport, defaults to canvas width
 10  * @property {int} max_y  Maximum y-position for viewport, defaults to canvas height 
 11  * @property {int} x      X-position for the upper left corner of the viewport
 12  * @property {int} y      Y-position for the upper left corner of the viewport
 13  *
 14  * @example
 15  * // Center viewport around players position (player needs to have x/y attributes)
 16  * // Usefull for sidescrollers
 17  * viewport.centerAround(player)
 18  *
 19  * // Common viewport usage. max_x/max_y could be said to set the "game world size"
 20  * viewport = viewport = new jaws.Viewport({max_x: 400, max_y: 3000})
 21  * player = new jaws.Sprite({x:100, y:400})
 22  * viewport.centerAround(player)
 23  *
 24  * // Draw player relative to the viewport. If viewport is way off, player won't even show up.
 25  * viewport.apply( function() {
 26  *  player.draw()
 27  * });
 28  *
 29  */
 30 jaws.Viewport = function ViewPort(options) {
 31   if( !(this instanceof arguments.callee) ) return new arguments.callee( options );
 32 
 33   this.options = options
 34   this.context = options.context || jaws.context
 35   this.width = options.width || jaws.width
 36   this.height = options.height || jaws.height
 37   this.max_x = options.max_x || jaws.width 
 38   this.max_y = options.max_y || jaws.height
 39   this.x = options.x || 0
 40   this.y = options.y || 0
 41   var that = this
 42 
 43   /** Move viewport x pixels horizontally and y pixels vertically */
 44   this.move = function(x, y) {
 45     x && (this.x += x)
 46     y && (this.y += y)
 47     this.verifyPosition()
 48   };
 49   
 50   /** Move viewport to given x/y */
 51   this.moveTo = function(x, y) {
 52     if(!(x==undefined)) { this.x = x }
 53     if(!(y==undefined)) { this.y = y }
 54     this.verifyPosition()
 55   };
 56 
 57   /** 
 58    * Returns true if item is outside viewport 
 59    * @example
 60    *
 61    * if( viewport.isOutside(player)) player.die();
 62    *
 63    * // ... or the more advanced:
 64    * bullets = new SpriteList()
 65    * bullets.push( bullet )
 66    * bullets.removeIf( viewport.isOutside )
 67    *
 68    */
 69   this.isOutside = function(item) {
 70     return(!that.isInside(item))
 71   };
 72 
 73   /** Returns true if item is inside viewport  */
 74   this.isInside = function(item) {
 75     return( item.x >= that.x && item.x <= (that.x + that.width) && item.y >= that.y && item.y <= (that.y + that.height) )
 76   };
 77 
 78   /** Returns true if item is partly (down to 1 pixel) inside viewport */
 79   this.isPartlyInside = function(item) {
 80     var rect = item.rect()
 81     return( rect.right >= that.x && rect.x <= (that.x + that.width) && rect.bottom >= that.y && item.y <= (that.y + that.height) )
 82   };
 83   
 84   /** Returns true of item is left of viewport */
 85   this.isLeftOf = function(item) { return(item.x < that.x)  }
 86  
 87   /** Returns true of item is right of viewport */
 88   this.isRightOf = function(item) { return(item.x > (that.x + that.width) )  }
 89 
 90   /** Returns true of item is above viewport */
 91   this.isAbove = function(item) { return(item.y < that.y)  }
 92 
 93   /** Returns true of item is above viewport */
 94   this.isBelow = function(item) { return(item.y > (that.y + that.height) )  }
 95 
 96 
 97   /** 
 98    * center the viewport around item. item must respond to x and y for this to work. 
 99    * Usefull for sidescrollers when you wan't to keep the player in the center of the screen no matter how he moves.
100    */
101   this.centerAround = function(item) {
102     this.x = Math.floor(item.x - this.width / 2);
103     this.y = Math.floor(item.y - this.height / 2);
104     this.verifyPosition();
105   };
106 
107   /**
108    * force 'item' inside current viewports visible area
109    * using 'buffer' as indicator how close to the 'item' is allowed to go
110    *
111    * @example
112    *
113    * viewport.move(10,0)                          // scroll forward
114    * viewport.forceInsideVisibleArea(player, 20)  // make sure player doesn't get left behind
115    */
116   this.forceInsideVisibleArea = function(item, buffer) {
117     if(item.x < this.x+buffer)               { item.x = this.x+buffer }
118     if(item.x > this.x+jaws.width-buffer)    { item.x = this.x+jaws.width-buffer }
119     if(item.y < this.y+buffer)               { item.y = this.y+buffer }
120     if(item.y > this.y+jaws.height-buffer)   { item.y = this.y+jaws.height-buffer }
121   }
122   
123   /**
124    * force 'item' inside the limits of the viewport
125    * using 'buffer' as indicator how close to the 'item' is allowed to go
126    *
127    * @example
128    * viewport.forceInside(player, 10) 
129    */
130   this.forceInside = function(item, buffer) {
131     if(item.x < buffer)               { item.x = buffer }
132     if(item.x > this.max_x-buffer)    { item.x = this.max_x-buffer }
133     if(item.y < buffer)               { item.y = buffer }
134     if(item.y > this.max_y-buffer)    { item.y = this.max_y-buffer }
135   }
136 
137 
138   /** 
139   * executes given draw-callback with a translated canvas which will draw items relative to the viewport
140   * 
141   * @example
142   *
143   * viewport.apply( function() {
144   *   player.draw();
145   *   foo.draw();
146   * });
147   * 
148   */
149   this.apply = function(func) {
150     this.context.save()
151     this.context.translate(-this.x, -this.y)
152     func()
153     this.context.restore()
154   };
155 
156   /** 
157    * if obj is an array-like object, iterate through it and call draw() on each item if it's partly inside the viewport 
158    */
159   this.draw = function( obj ) {
160     this.apply( function() {
161       if(obj.forEach) obj.forEach( that.drawIfPartlyInside );
162       else if(obj.draw) that.drawIfPartlyInside(obj);
163       // else if(jaws.isFunction(obj) {};  // add apply()-functionally here?
164     });
165   }
166 
167   /** 
168    * draws all items of 'tile_map' that's lies inside the viewport 
169    * this is simular to viewport.draw( tile_map.all() ) but optmized for Huge game worlds (tile maps)
170    */
171   this.drawTileMap = function( tile_map ) {
172     var sprites = tile_map.atRect({ x: this.x, y: this.y, right: this.x + this.width, bottom: this.y + this.height })
173     this.apply( function() {
174       for(var i=0; i < sprites.length; i++) sprites[i].draw();
175     });
176   }
177 
178   /** draws 'item' if it's partly inside the viewport */
179   this.drawIfPartlyInside = function(item) { 
180     if(that.isPartlyInside(item)) item.draw(); 
181   }
182 
183   /** @private */
184   this.verifyPosition = function() {
185     var max = this.max_x - this.width
186     if(this.x < 0)      { this.x = 0 }
187     if(this.x > max)    { this.x = max }
188 
189     var max = this.max_y - this.height
190     if(this.y < 0)      { this.y = 0 }
191     if(this.y > max)    { this.y = max }
192   };
193  
194   this.moveTo(options.x||0, options.y||0)
195 }
196 
197 jaws.Viewport.prototype.toString = function() { return "[Viewport " + this.x.toFixed(2) + ", " + this.y.toFixed(2) + ", " + this.width + ", " + this.height + "]" }
198 
199 return jaws;
200 })(jaws || {});
