  3 /** @private
  4  * Cut out a rectangular piece of a an image, returns as canvas-element 
  5  */
  6 function cutImage(image, x, y, width, height) {
  7   var cut = document.createElement("canvas")
  8   cut.width = width
  9   cut.height = height
 10   
 11   var ctx = cut.getContext("2d")
 12   ctx.drawImage(image, x, y, width, height, 0, 0, cut.width, cut.height)
 13   
 14   return cut
 15 };

 17 /** 
 18  * @class Cut out invidual frames (images) from a larger spritesheet-image. "Field Summary" contains options for the SpriteSheet()-constructor.
 19  *
 20  * @property {image|image} Image/canvas or asset-string to cut up smaller images from
 21  * @property {string} orientation How to cut out invidual images from spritesheet, either "right" or "down"
 22  * @property {array} frame_size  width and height of invidual frames in spritesheet
 23  * @property {array} frames all single frames cut out from image
 24  * @property {integer} offset vertical or horizontal offset to start cutting from
 25  * @property {int} scale_image Scale the sprite sheet by this factor before cutting out the frames. frame_size is automatically re-sized too
 26  *
 27 */
 28 jaws.SpriteSheet = function SpriteSheet(options) {
 29   if( !(this instanceof arguments.callee) ) return new arguments.callee( options );
 30 
 31   this.image = jaws.isDrawable(options.image) ? options.image : jaws.assets.data[options.image]
 32   this.orientation = options.orientation || "down"
 33   this.frame_size = options.frame_size || [32,32]
 34   this.frames = []
 35   this.offset = options.offset || 0
 36   
 37   if(options.scale_image) {
 38     var image = (jaws.isDrawable(options.image) ? options.image : jaws.assets.get(options.image))
 39     this.frame_size[0] *= options.scale_image
 40     this.frame_size[1] *= options.scale_image
 41     options.image = jaws.gfx.retroScaleImage(image, options.scale_image)
 42   }
 43 
 44   var index = 0
 45 
 46   // Cut out tiles from Top -> Bottom
 47   if(this.orientation == "down") {  
 48     for(var x=this.offset; x < this.image.width; x += this.frame_size[0]) {
 49       for(var y=0; y < this.image.height; y += this.frame_size[1]) {
 50         this.frames.push( cutImage(this.image, x, y, this.frame_size[0], this.frame_size[1]) )
 51       }
 52     }
 53   }
 54   // Cut out tiles from Left -> Right
 55   else {
 56     for(var y=this.offset; y < this.image.height; y += this.frame_size[1]) {
 57       for(var x=0; x < this.image.width; x += this.frame_size[0]) {
 58         this.frames.push( cutImage(this.image, x, y, this.frame_size[0], this.frame_size[1]) )
 59       }
 60     }
 61   }
 62 }
