  3 /**
  4   @class A Basic rectangle.
  5   @example
  6   rect = new jaws.Rect(5,5,20,20)
  7   rect.right  // -> 25
  8   rect.bottom // -> 25
  9   rect.move(10,20)
 10   rect.right  // -> 35
 11   rect.bottom // -> 45
 12   rect.width  // -> 20
 13   rect.height // -> 20
 14 */
 15 jaws.Rect = function Rect(x, y, width, height) {
 16   if( !(this instanceof arguments.callee) ) return new arguments.callee(x, y, width, height);
 17   
 18   this.x = x
 19   this.y = y
 20   this.width = width
 21   this.height = height
 22   this.right = x + width
 23   this.bottom = y + height
 24 }
 25 
 26 /** Return position as [x,y] */
 27 jaws.Rect.prototype.getPosition = function() {
 28   return [this.x, this.y]
 29 }

 31 /** Move rect x pixels horizontally and y pixels vertically */
 32 jaws.Rect.prototype.move = function(x,y) {
 33   this.x += x
 34   this.y += y
 35   this.right += x
 36   this.bottom += y
 37   return this
 38 }

 46   return this
 47 }
