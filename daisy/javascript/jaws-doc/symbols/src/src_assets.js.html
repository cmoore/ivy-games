1 var jaws = (function(jaws) {
  2 
  3 /**
  4  * @class Loads and processes assets as images, sound, video, json
  5  * Used internally by JawsJS to create <b>jaws.assets</b>
  6  *
  7  * @property {bool} bust_cache              Add random arguments to assets-url to bypass any cache
  8  * @property {bool} fuchia_to_transparent   Convert the color fuchia to transparent when loading .bmp-files
  9  * @proparty {bool} image_to_canvas         Convert all image assets to canvas internally
 10  * @proparty {string} root                  Rootdir from where all assets are loaded
 11  *
 12  */
 13 jaws.Assets = function Assets() {
 14   if( !(this instanceof arguments.callee) ) return new arguments.callee();
 15 
 16   this.loaded = []    // Hash of all URLs that's been loaded
 17   this.loading = []   // Hash of all URLs currently loading
 18   this.src_list = []  // Hash of all unloaded URLs that loadAll() will try to load
 19   this.data = []      // Hash of loaded raw asset data, URLs are keys
 20 
 21   this.bust_cache = false
 22   this.image_to_canvas = true
 23   this.fuchia_to_transparent = true
 24   this.root = ""
 25 
 26   this.file_type = {}
 27   this.file_type["json"] = "json"
 28   this.file_type["wav"] = "audio"
 29   this.file_type["mp3"] = "audio"
 30   this.file_type["ogg"] = "audio"
 31   this.file_type["png"] = "image"
 32   this.file_type["jpg"] = "image"
 33   this.file_type["jpeg"] = "image"
 34   this.file_type["gif"] = "image"
 35   this.file_type["bmp"] = "image"
 36   this.file_type["tiff"] = "image"
 37   var that = this
 38 
 39   this.length = function() {
 40     return this.src_list.length
 41   }
 42 
 43   /*
 44    * Get one or many resources
 45    *
 46    * @param   String or Array of strings
 47    * @returns The raw resource or an array of resources
 48    *
 49    */
 50   this.get = function(src) {
 51     if(jaws.isArray(src)) {
 52       return src.map( function(i) { return that.data[i] } )
 53     }
 54     else {
 55       if(this.loaded[src])  { return this.data[src] }
 56       else                  { jaws.log("No such asset: " + src, true) }
 57     }
 58   }
 59 
 60   /** Return true if src is in the process of loading (but not yet finishing) */
 61   this.isLoading = function(src) {
 62     return this.loading[src]
 63   }
 64 
 65   /** Return true if src is loaded in full */
 66   this.isLoaded = function(src) {
 67     return this.loaded[src]
 68   }
 69 
 70   this.getPostfix = function(src) {
 71     postfix_regexp = /\.([a-zA-Z0-9]+)/;
 72     return postfix_regexp.exec(src)[1]
 73   }
 74 
 75   this.getType = function(src) {
 76     var postfix = this.getPostfix(src)
 77     return (this.file_type[postfix] ? this.file_type[postfix] : postfix)
 78   }
 79 
 80   /**
 81    * Add array of paths or single path to asset-list. Later load with loadAll()
 82    *
 83    * @example
 84    *
 85    * jaws.assets.add("player.png")
 86    * jaws.assets.add(["media/bullet1.png", "media/bullet2.png"])
 87    * jaws.assets.loadAll({onfinish: start_game})
 88    *
 89    */
 90   this.add = function(src) {
 91     if(jaws.isArray(src)) { for(var i=0; src[i]; i++) { this.add(src[i]) } }
 92     else                  { this.src_list.push(src) }
 93     // else                  { var path = this.root + src; this.src_list.push(path) }
 94     return this
 95   }
 96 
 97   /** Load all pre-specified assets */
 98   this.loadAll = function(options) {
 99     this.load_count = 0
100     this.error_count = 0
101 
102     /* With these 3 callbacks you can display progress and act when all assets are loaded */
103     this.onload = options.onload
104     this.onerror = options.onerror
105     this.onfinish = options.onfinish
106 
107     for(i=0; this.src_list[i]; i++) {
108       this.load(this.src_list[i])
109     }
110   }
111 
112   /** Calls onload right away if asset is available since before, otherwise try to load it */
113   this.getOrLoad = function(src, onload, onerror) {
114     if(this.data[src]) { onload() }
115     else { this.load(src, onload, onerror) }
116   }
117 
118   /** 
119    * Load a single url <b>src</b>.
120    * if <b>onload</b> is specified, it's called on loading-success
121    * if <b>onerror</b> is specified, it will be called on any loading-error
122    * 
123    * @example
124    *
125    *   jaws.load("media/foo.png")
126    *
127    */
128   this.load = function(src, onload, onerror) {
129     var asset = {}
130     asset.src = src
131     asset.onload = onload
132     asset.onerror = onerror
133     this.loading[src] = true
134 
135     var resolved_src = this.root + asset.src;
136     if (this.bust_cache) { resolved_src += "?" + parseInt(Math.random()*10000000) }
137 
138     switch(this.getType(asset.src)) {
139       case "image":
140         asset.image = new Image()
141         asset.image.asset = asset // enables us to access asset in the callback
142         //
143         // TODO: Make http://dev.ippa.se/webgames/test2.html work
144         //
145         asset.image.onload = this.assetLoaded
146         asset.image.onerror = this.assetError
147         asset.image.src = resolved_src
148         break;
149       case "audio":
150         asset.audio = new Audio(resolved_src)
151         asset.audio.asset = asset         // enables us to access asset in the callback
152         this.data[asset.src] = asset.audio
153         asset.audio.addEventListener("canplay", this.assetLoaded, false);
154         asset.audio.addEventListener("error", this.assetError, false);
155         asset.audio.load()
156         break;
157       default:
158         var req = new XMLHttpRequest()
159         req.asset = asset         // enables us to access asset in the callback
160         req.onreadystatechange = this.assetLoaded
161         req.open('GET', resolved_src, true)
162         req.send(null)
163         break;
164     }
165   }
166 
167   /** @private
168    * Callback for all asset-loading.
169    * 1) Parse data depending on filetype. Images are (optionally) converted to canvas-objects. json are parsed into native objects and so on.
170    * 2) Save processed data in internal list for easy fetching with assets.get(src) later on
171    * 3) Call callbacks if defined
172    */
173   this.assetLoaded = function(e) {
174     var asset = this.asset
175     var src = asset.src
176     var filetype = that.getType(asset.src)
177 
178     // Keep loading and loaded hash up to date
179     that.loaded[src] = true
180     that.loading[src] = false
181 
182     // Process data depending differently on postfix
183     if(filetype == "json") {
184       if (this.readyState != 4) { return }
185       that.data[asset.src] = JSON.parse(this.responseText)
186     }
187     else if(filetype == "image") {
188       var new_image = that.image_to_canvas ? jaws.imageToCanvas(asset.image) : asset.image
189       if(that.fuchia_to_transparent && that.getPostfix(asset.src) == "bmp") { new_image = fuchiaToTransparent(new_image) }
190       that.data[asset.src] = new_image
191     }
192     else if(filetype == "audio") {
193       asset.audio.removeEventListener("canplay", that.assetLoaded, false);
194       that.data[asset.src] = asset.audio
195     }
196 
197     that.load_count++
198     that.processCallbacks(asset, true)
199   }
200 
201   /** @private */
202   this.assetError = function(e) {
203     var asset = this.asset
204     that.error_count++
205     that.processCallbacks(asset, false)
206   }
207 
208   /** @private */
209   this.processCallbacks = function(asset, ok) {
210     var percent = parseInt( (that.load_count+that.error_count) / that.src_list.length * 100)
211 
212     if(ok) {
213       if(that.onload)   that.onload(asset.src, percent);
214       if(asset.onload)  asset.onload();
215     }
216     else {
217       if(that.onerror)  that.onerror(asset.src, percent);
218       if(asset.onerror) asset.onerror(asset);
219     }
220 
221     // When loadAll() is 100%, call onfinish() and kill callbacks (reset with next loadAll()-call)
222     if(percent==100) {
223       if(that.onfinish) { that.onfinish() }
224       that.onload = null
225       that.onerror = null
226       that.onfinish = null
227     }
228   }
229 }

231 /** @private
232  * Make Fuchia (0xFF00FF) transparent
233  * This is the de-facto standard way to do transparency in BMPs
234  * Returns: a canvas-element
235  */
236 function fuchiaToTransparent(image) {
237   canvas = jaws.isImage(image) ? jaws.imageToCanvas(image) : image
238   var context = canvas.getContext("2d")
239   var img_data = context.getImageData(0,0,canvas.width,canvas.height)
240   var pixels = img_data.data
241   for(var i = 0; i < pixels.length; i += 4) {
242     if(pixels[i]==255 && pixels[i+1]==0 && pixels[i+2]==255) { // Color: Fuchia
243       pixels[i+3] = 0 // Set total see-through transparency
244     }
245   }
246   context.putImageData(img_data,0,0);
247   return canvas
248 }
249 
250 jaws.assets = new jaws.Assets()
251 return jaws;
252 })(jaws || {});
