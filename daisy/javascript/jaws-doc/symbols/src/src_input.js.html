105 /** @private
106  * handle event "onkeydown" by remembering what key was pressed
107  */
108 function handleKeyUp(e) {
109   event = (e) ? e : window.event
110   var human_name = keycode_to_string[event.keyCode]
111   pressed_keys[human_name] = false
112   if(on_keyup_callbacks[human_name]) { 
113     on_keyup_callbacks[human_name](human_name)
114     e.preventDefault()
115   }
116   if(prevent_default_keys[human_name]) { e.preventDefault() }
117 }
118 
119 /** @private
120  * handle event "onkeydown" by remembering what key was un-pressed
121  */
122 function handleKeyDown(e) {
123   event = (e) ? e : window.event  
124   var human_name = keycode_to_string[event.keyCode]
125   pressed_keys[human_name] = true
126   if(on_keydown_callbacks[human_name]) { 
127     on_keydown_callbacks[human_name](human_name)
128     e.preventDefault()
129   }
130   if(prevent_default_keys[human_name]) { e.preventDefault() }
131 }
132 /** @private
133  * handle event "onmousedown" by remembering what button was pressed

135 function handleMouseDown(e) {

138   if (navigator.appName == "Microsoft Internet Explorer"){
139 	  human_name = ie_mousebuttoncode_to_string[event.button];

141   pressed_keys[human_name] = true
142   if(on_keydown_callbacks[human_name]) { 
143     on_keydown_callbacks[human_name](human_name)
144     e.preventDefault()
145   }

166 /** @private
167  * handle event "touchstart" by remembering what button was pressed
168  */
169 function handleTouchStart(e) {
170 	event = (e) ? e : window.event  
171 	pressed_keys["left_mouse_button"] = true
172 	jaws.mouse_x = e.touches[0].pageX - jaws.canvas.offsetLeft;
173 	jaws.mouse_y = e.touches[0].pageY - jaws.canvas.offsetTop;
174 	//e.preventDefault()
175 }
176 
177 /** @private
178  * handle event "touchend" by remembering what button was pressed
179  */
180 function handleTouchEnd(e) {
181   event = (e) ? e : window.event  
182   pressed_keys["left_mouse_button"] = false
183 	jaws.mouse_x = undefined;
184 	jaws.mouse_y = undefined;
185 
186 }

188 var prevent_default_keys = []
189 /** 
190  * Prevents default browseraction for given keys.
191  * @example
192  * jaws.preventDefaultKeys( ["down"] )  // Stop down-arrow-key from scrolling page down
193  */
194 jaws.preventDefaultKeys = function(array_of_strings) {
195   array_of_strings.forEach( function(item, index) {
196     prevent_default_keys[item] = true
197   });
198 }

200 /**
201  * Returns true if *key* is currently pressed down
202  * @example
203  * jaws.pressed("left");  // returns true if arrow key is pressed
204  * jaws.pressed("a");     // returns true if key "a" is pressed
205  */
206 jaws.pressed = function(key) {
207   return pressed_keys[key]
208 }

210 /** 
211  * sets up a callback for a key (or array of keys) to call when it's pressed down
212  * 
213  * @example
214  * // call goLeft() when left arrow key is  pressed
215  * jaws.on_keypress("left", goLeft) 
216  *
217  * // call fireWeapon() when SPACE or CTRL is pressed
218  * jaws.on_keypress(["space","ctrl"], fireWeapon)
219  */
220 jaws.on_keydown = function(key, callback) {
221   if(jaws.isArray(key)) {
222     for(var i=0; key[i]; i++) {
223       on_keydown_callbacks[key[i]] = callback
224     }
225   }
226   else {
227     on_keydown_callbacks[key] = callback
228   }
229 }
230 
231 /** 
232  * sets up a callback when a key (or array of keys) to call when it's released 
233  */
234 jaws.on_keyup = function(key, callback) {
235   if(jaws.isArray(key)) {
236     for(var i=0; key[i]; i++) {
237       on_keyup_callbacks[key[i]] = callback
238     }
239   }
240   else {
241     on_keyup_callbacks[key] = callback
242   }
243 }
