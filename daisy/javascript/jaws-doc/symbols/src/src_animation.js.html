1 var jaws = (function(jaws) {
  2 
  3 /**
  4  * @class Manages an animation with a given list of frames. "Field Summary" contains options for the Animation()-constructor.
  5  *
  6  * @property {bool} loop        Restart animation when end is reached
  7  * @property {bool} bounce      Rewind the animation frame by frame when end is reached
  8  * @property {int} index          Start on this frame
  9  * @property {array} frames       Images/canvaselements
 10  * @property {milliseconds} frame_duration  How long should each frame be displayed
 11  * @property {int} frame_direction  -1 for backwards animation. 1 is default
 12  * @property {array} frame_size     Containing width/height, eg. [32, 32]
 13  * @property {int} offset           When cutting out frames from a sprite sheet, start at this frame
 14  * @property {string} orientation   How to cut out frames frmo sprite sheet, possible values are "down" or "right"
 15  * @property {function} on_end      Function to call when animation ends. triggers only on non-looping, non-bouncing animations
 16  *
 17  * @example
 18  * // in setup()
 19  * anim = new jaws.Animation({sprite_sheet: "droid_11x15.png", frame_size: [11,15], frame_duration: 100})
 20  * player = new jaws.Sprite({y:300, anchor: "center_bottom"})
 21  *
 22  * // in update()
 23  * player.setImage( anim.next() )
 24  *
 25  * // in draw()
 26  * player.draw()
 27  *
 28  */
 29 jaws.Animation = function Animation(options) {
 30   if( !(this instanceof arguments.callee) ) return new arguments.callee( options );
 31 
 32   this.options = options
 33   this.frames = options.frames || []
 34   this.frame_duration = options.frame_duration || 100   // default: 100ms between each frameswitch
 35   this.index = options.index || 0                       // default: start with the very first frame
 36   this.loop = (options.loop==undefined) ? 1 : options.loop
 37   this.bounce = options.bounce || 0
 38   this.frame_direction = 1
 39   this.frame_size = options.frame_size
 40   this.orientation = options.orientation || "down"
 41   this.on_end = options.on_end || null
 42   this.offset = options.offset || 0
 43 
 44   if(options.scale_image) {
 45     var image = (jaws.isDrawable(options.sprite_sheet) ? options.sprite_sheet : jaws.assets.get(options.sprite_sheet))
 46     this.frame_size[0] *= options.scale_image
 47     this.frame_size[1] *= options.scale_image
 48     options.sprite_sheet = jaws.gfx.retroScaleImage(image, options.scale_image)
 49   }
 50 
 51   if(options.sprite_sheet) {
 52     var image = (jaws.isDrawable(options.sprite_sheet) ? options.sprite_sheet : jaws.assets.get(options.sprite_sheet))
 53     var sprite_sheet = new jaws.SpriteSheet({image: image, frame_size: this.frame_size, orientation: this.orientation, offset: this.offset})
 54     this.frames = sprite_sheet.frames
 55   }
 56 
 57   /* Initializing timer-stuff */
 58   this.current_tick = (new Date()).getTime();
 59   this.last_tick = (new Date()).getTime();
 60   this.sum_tick = 0
 61 }
 62 
 63 /**
 64  Propells the animation forward by counting milliseconds and changing this.index accordingly
 65  Supports looping and bouncing animations
 66 */
 67 jaws.Animation.prototype.update = function() {
 68   this.current_tick = (new Date()).getTime();
 69   this.sum_tick += (this.current_tick - this.last_tick);
 70   this.last_tick = this.current_tick;
 71 
 72   if(this.sum_tick > this.frame_duration) {
 73     this.index += this.frame_direction
 74     this.sum_tick = 0
 75   }
 76   if( (this.index >= this.frames.length) || (this.index < 0) ) {
 77     if(this.bounce) {
 78       this.frame_direction = -this.frame_direction
 79       this.index += this.frame_direction * 2
 80     }
 81     else if(this.loop) {
 82       this.index = 0
 83     }
 84     else {
 85       this.index -= this.frame_direction
 86       if (this.on_end) {
 87         this.on_end()
 88         this.on_end = null
 89       }
 90     }
 91   }
 92   return this
 93 }
 94 
 95 /**
 96   works like Array.slice but returns a new Animation-object with a subset of the frames
 97 */
 98 jaws.Animation.prototype.slice = function(start, stop) {
 99   var o = {}
100   o.frame_duration = this.frame_duration
101   o.loop = this.loop
102   o.bounce = this.bounce
103   o.on_end = this.on_end
104   o.frame_direction = this.frame_direction
105   o.frames = this.frames.slice().slice(start, stop)
106   return new jaws.Animation(o)
107 };
108 
109 /**
110   Moves animation forward by calling update() and then return the current frame
111 */
112 jaws.Animation.prototype.next = function() {
113   this.update()
114   return this.frames[this.index]
115 };
116 
117 /** returns true if animation is at the very last frame */
118 jaws.Animation.prototype.atLastFrame = function() { return (this.index == this.frames.length-1) }
119 
120 /** returns true if animation is at the very first frame */
121 jaws.Animation.prototype.atFirstFrame = function() { return (this.index == 0) }
122 
123 
124 /**
125   returns the current frame
126 */
127 jaws.Animation.prototype.currentFrame = function() {
128   return this.frames[this.index]
129 };
130 
131 /**
132  * Debugstring for Animation()-constructor
133  * @example
134  * var anim = new Animation(...)
135  * console.log(anim.toString())
136  */
137 jaws.Animation.prototype.toString = function() { return "[Animation, " + this.frames.length + " frames]" }
138 
139 return jaws;
140 })(jaws || {});
