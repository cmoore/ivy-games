 15 /**
 16  * @class A classic game loop forever looping calls to update() / draw() with given framerate. "Field Summary" contains options for the GameLoop()-constructor.
 17  *
 18  * @property {int} tick_duration  duration in ms between the last 2 ticks (often called dt)
 19  * @property {int} fps  the real fps (as opposed to the target fps), smoothed out with a moving average
 20  * @property {int} ticks  total amount of ticks since game loops start
 21  *
 22  * @example
 23  *
 24  * game = {}
 25  *  draw: function() { ... your stuff executed every 30 FPS ... }
 26  * }
 27  *
 28  * game_loop = new jaws.GameLoop(game, {fps: 30})
 29  * game_loop.start()
 30  *
 31  * // You can also use the shortcut jaws.start(), it will:
 32  * // 1) Load all assets with jaws.assets.loadAll()
 33  * // 2) Create a GameLoop() and start it
 34  * jaws.start(MyGameState, {fps: 30})
 35  *
 36  */
 37 jaws.GameLoop = function GameLoop(game_object, options,game_state_setup_options) {
 38   if( !(this instanceof arguments.callee) ) return new arguments.callee( game_object, options );
 39 
 40   this.tick_duration = 0
 41   this.fps = 0
 42   this.ticks = 0
 43   
 44   var update_id
 45   var paused = false
 46   var stopped = false
 47   var that = this
 48   var mean_value = new MeanValue(20) // let's have a smooth, non-jittery FPS-value
 49 
 50   /** 
 51    * returns how game_loop has been active in milliseconds 
 52    * does currently not factor in pause-time
 53    */
 54   this.runtime = function() {
 55     return (this.last_tick - this.first_tick)
 56   }
 57 
 58   /** Start the game loop by calling setup() once and then loop update()/draw() forever with given FPS */
 59   this.start = function() {
 60     jaws.log("game loop start", true)
 61   
 62     this.first_tick = (new Date()).getTime();
 63     this.current_tick = (new Date()).getTime();
 64     this.last_tick = (new Date()).getTime(); 
 65 
 66     if(game_object.setup) { game_object.setup(game_state_setup_options) }
 67     step_delay = 1000 / options.fps;
 68    
 69     if(options.fps == 60) {
 70       requestAnimFrame(this.loop)
 71     }
 72     else {
 73       update_id = setInterval(this.loop, step_delay);
 74     }
 75 
 76     jaws.log("game loop loop", true)
 77   }
 78   
 79   /** The core of the game loop. Calculate a mean FPS and call update()/draw() if game loop is not paused */
 80   this.loop = function() {
 81     that.current_tick = (new Date()).getTime();
 82     that.tick_duration = that.current_tick - that.last_tick
 83     that.fps = mean_value.add(1000/that.tick_duration).get()
 84 
 85     if(!stopped && !paused) {
 86       if(game_object.update) { game_object.update() }
 87       if(game_object.draw)   { game_object.draw() }
 88       that.ticks++
 89     }
 90     if(options.fps == 60 && !stopped) requestAnimFrame(that.loop);
 91     that.last_tick = that.current_tick;
 92   }
 93   
 94   /** Pause the game loop. loop() will still get called but not update() / draw() */
 95   this.pause = function()   { paused = true }
 96   
 97   /** unpause the game loop */
 98   this.unpause = function() { paused = false }
 99 
100   /** Stop the game loop */
101   this.stop = function() { 
102     if(update_id) clearInterval(update_id); 
103     stopped = true;
104   }
105 }

107 /** @ignore */
108 function MeanValue(size) {
109   this.size = size
110   this.values = new Array(this.size)
111   this.value
112   
113   this.add = function(value) {
114     if(this.values.length > this.size) {  // is values filled?
115       this.values.splice(0,1)
116       this.value = 0
117       for(var i=0; this.values[i]; i++) {
118         this.value += this.values[i]
119       }
120       this.value = this.value / this.size
121     }
122     this.values.push(value)
123     
124     return this
125   }
126 
127   this.get = function() {
128     return parseInt(this.value)
129   }

131 }
